import { Injectable } from '@nestjs/common'; import { QueryFailedError } from 'typeorm'; import { DomainErrors } from '@/common/domain/error/domain-errors.factory'; import { IAppError } from '@/common/domain/error/app-error.interface'; /** * Maps raw database errors to domain-level IAppError objects. * * This is the ONLY place in the codebase that knows about PostgreSQL * error codes and their meaning. Services and controllers remain * completely unaware of database internals. */ @Injectable() export class DatabaseErrorMapper { /** * Translates a caught DB error into a typed IAppError. * @param error The raw error caught from TypeORM * @param entity Human-readable entity name for dynamic messages */ public map(error: unknown, entity: string): IAppError { if (error instanceof QueryFailedError) { const err = error as QueryFailedError & { code?: string; detail?: string; }; // ── PostgreSQL 23505: unique_violation ────────────────────────── if (err.code === '23505') { const match = err.detail?.match(/\((.*?)\)=\((.*?)\)/); const field = match?.[1] ?? 'unknown'; const value = match?.[2] ?? 'unknown'; return DomainErrors.alreadyExists({ entity, field, value }); } // ── PostgreSQL 23503: foreign_key_violation ───────────────────── if (err.code === '23503') { return DomainErrors.internalError({ action: `save ${entity}`, reason: 'Referenced record not found (foreign key violation)', }); } // ── PostgreSQL 23502: not_null_violation ──────────────────────── if (err.code === '23502') { return DomainErrors.internalError({ action: `save ${entity}`, reason: 'A required field is missing', }); } } return DomainErrors.internalError({ action: `database operation on ${entity}` }); } }

'Error', status, detail, instance, errorCode, }; if (validationErrors) body.validationErrors = validationErrors; response.status(status).json(body); } // ── Unknown / unhandled ──────────────────────────────────────────────────── private handleUnknown( exception: unknown, response: Response, instance: string, ): void { this.logger.error('Unhandled exception', exception); const slug = DomainErrorCode.INTERNAL_ERROR.toLowerCase().replace(/_/g, '-'); response.status(HttpStatus.INTERNAL_SERVER_ERROR).json({ type: `${this.errorsBaseUrl}/${slug}`, title: 'Internal Server Error', status: HttpStatus.INTERNAL_SERVER_ERROR, detail: 'An unexpected error occurred', instance, errorCode: DomainErrorCode.INTERNAL_ERROR, } satisfies ProblemDetails); } // ── Helpers ──────────────────────────────────────────────────────────────── private statusToErrorCode(status: number): string { const map: Record<number, string> = { 400: 'VALIDATION_FAILED', 401: 'UNAUTHORIZED', 403: 'ACCESS_DENIED', 404: 'RESOURCE_NOT_FOUND', 409: 'CONFLICT', 422: 'VALIDATION_FAILED', 500: 'INTERNAL_ERROR', }; return map[status] ?? import { CallHandler, ExecutionContext, Injectable, NestInterceptor, } from '@nestjs/common'; import { Observable } from 'rxjs'; import { mergeMap } from 'rxjs/operators'; /** * Global interceptor that unwraps neverthrow Result / ResultAsync values * returned by controller handlers. * * - If the resolved value is a neverthrow Result (duck-typed via `isOk`): * - `isOk()` → returns `value` (clean data) * - `isErr()` → throws `error` (IAppError), caught by DomainExceptionFilter * - Otherwise, the value is passed through unchanged. * * This eliminates the need for `throwDomainError` helpers in controllers. * * NOTE: Controllers using `@Res({ passthrough: true })` bypass NestJS * response transformation, so this interceptor won't affect them. * Those controllers (e.g., AuthController) must still unwrap Results manually. */ @Injectable() export class ResultInterceptor implements NestInterceptor { intercept(_context: ExecutionContext, next: CallHandler): Observable<unknown> { return next.handle().pipe( mergeMap(async (value: unknown) => { // Support for ResultAsync, Promise<Result>, and synchronous Result. // We use duck typing or direct await because ResultAsync is PromiseLike but NOT instanceof Promise. const resolvedValue = value && typeof (value as any).then === 'function' ? await value : value; // Duck typing to check if the object is a neverthrow Result if ( resolvedValue !== null && resolvedValue !== undefined && typeof resolvedValue === 'object' && 'isOk' in resolvedValue && typeof (resolvedValue as Record<string, unknown>).isOk === 'function' ) { const result = resolvedValue as { isOk: () => boolean; isErr: () => boolean; value: unknown; error: unknown; }; if (result.isErr()) { throw result.error; // IAppError → caught by DomainExceptionFilter } return result.value; // Clean data } return resolvedValue; }), ); } }
